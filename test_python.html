PYTHON WEB SANDBOX

Môi trường kiểm thử code. Hỗ trợ Load File từ thiết bị.

INPUT: main.py Ngôn ngữ: Python 3
OUTPUT: Màn hình Console
Kết quả sẽ hiển thị ở đây... (Lần chạy đầu tiên có thể mất vài giây)
import sys import io from js import document def execute_python(*args, **kwargs): code = document.getElementById('pythonCode').value out_element = document.getElementById('output') out_element.innerHTML = "Đang xử lý..." old_stdout = sys.stdout sys.stdout = mystdout = io.StringIO() try: exec(code) result = mystdout.getvalue() if result.strip() != "": out_element.innerHTML = f"{result}" else: out_element.innerHTML = "[Code chạy thành công nhưng không có lệnh in (print) ra màn hình]" except Exception as e: out_element.innerHTML = f"Lỗi cú pháp / Lỗi thực thi:\n{str(e)}" finally: sys.stdout = old_stdout

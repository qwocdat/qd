import sys import io from js import document def execute_python(*args, **kwargs): # Lấy code từ ô nhập liệu code = document.getElementById('pythonCode').value out_element = document.getElementById('output') out_element.innerHTML = "Đang xử lý..." # Chuyển hướng màn hình console (stdout) vào một biến ẩn old_stdout = sys.stdout sys.stdout = mystdout = io.StringIO() try: # Ép trình duyệt chạy đoạn code vừa dán exec(code) # Thu thập toàn bộ các lệnh print() trong đoạn code đó result = mystdout.getvalue() # In ra màn hình HTML if result.strip() != "": out_element.innerHTML = f"{result}" else: out_element.innerHTML = "[Code chạy thành công nhưng không có lệnh in ra màn hình]" except Exception as e: # Nếu code gõ sai, in ra lỗi màu đỏ out_element.innerHTML = f"Lỗi cú pháp:\n{str(e)}" finally: # Trả lại trạng thái mặc định cho trình duyệt sys.stdout = old_stdout
